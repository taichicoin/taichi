// 把 fetchAllGas 函数替换成这个 async function fetchGasQuick(address) { const res = await fetch( `https://api.bscscan.com/api` + `?module=account&action=txlist` + `&address=${address}` + `&endtimestamp=${CUTOFF_TIME}` + `&offset=1000` // 只查最近 1000 笔 + `&apikey=${API_KEY}` ); const data = await res.json(); let gasTotal = 0; if (data.result) { for (const tx of data.result) { if (tx.from && tx.from.toLowerCase() === address.toLowerCase() && tx.txreceipt_status == "1") { const gas = (Number(tx.gasUsed) * Number(tx.gasPrice)) / 1e18; gasTotal += gas; } } } return gasTotal; }
